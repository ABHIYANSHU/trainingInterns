using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading;

namespace ThreadPrograms
{
    class ThreadExample
    {
        public void Thread1() // Non Static Method
        {
            Thread t = Thread.CurrentThread;
            Console.WriteLine(t.Name + " is running");
            lock (this)
            {
                for (int i = 0; i < 10; i++)
                {
                    Console.WriteLine(i);
                    Thread.Sleep(500);//Thread Sleep
                }
            }
        }
        public static void Thread2() // Static Method
        {
            Thread t = Thread.CurrentThread;
            Console.WriteLine(t.Name + " is running");
            for (int i = 0; i < 10; i++)
            {
                Console.WriteLine(i);
                Thread.Sleep(500);
            }
        }
        

    }
    class MyProgram
    {
        public static void Main(String[] args)
        {
            Console.WriteLine("start of Main");
            ThreadExample p = new ThreadExample();
            Thread t1 = new Thread(new ThreadStart(p.Thread1));
            Thread t2 = new Thread(new ThreadStart(p.Thread1));// Call Non Static Method
            Thread t3 = new Thread(new ThreadStart(ThreadExample.Thread2));// Call Static Method
            t1.Name = "Player1";
            t2.Name = "Player2";
            t3.Name = "Player3";
            // t3.Priority = ThreadPriority.Highest;
            //t2.Priority = ThreadPriority.Normal;
            //t1.Priority = ThreadPriority.Lowest;
            t3.Priority = ThreadPriority.Lowest;//Thread Priority
            t2.Priority = ThreadPriority.Highest;
            t1.Priority = ThreadPriority.Normal;
            t1.Start();
            t1.Join();
            t3.Start();
            t2.Start();

            try {
                t1.Abort();//Thread Abort
                t2.Abort();
            }
            catch(ThreadAbortException e)
            {
                Console.WriteLine(e.Message);
            }
           // 

            Console.WriteLine("end of main");
            Console.ReadKey();
        }

    }
}
